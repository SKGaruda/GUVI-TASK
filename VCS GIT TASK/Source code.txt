# 2. INITIALIZE GIT REPOSITORY, STAGE AND COMMIT INITIAL FILES

git init
git status
git add .
git commit -m "Initial commit with script files"

----------------------------------------------------------------------------------------------------------------
# 3. CONNECT LOCAL REPO TO GITHUB

git remote add origin https://github.com/SKGaruda/Demotest1.git
git remote -v
git branch -M main
git push -u origin main

----------------------------------------------------------------------------------------------------------------
# 4. DEMONSTRATE GIT MERGE


git checkout -b feature-merge

echo "# Merge Demo" >> README.md
git add README.md
git commit -m "Feature branch: add merge demo"

git checkout main

git merge feature-merge

git push origin main

----------------------------------------------------------------------------------------------------------------
# 5. DEMONSTRATE GIT REBASE

git checkout -b feature-rebase

echo "Rebase Example" > rebase.txt
git add rebase.txt
git commit -m "Add rebase file"

git checkout main
echo "Main branch update" > main-update.txt
git add main-update.txt
git commit -m "Main branch update"

git checkout feature-rebase

git rebase main

git push --force-with-lease origin feature-rebase

git log --oneline --graph --all
----------------------------------------------------------------------------------------------------------------
# 6. DEMONSTRATE GIT STASH


echo "Temporary work" >> temp.txt

git stash

git stash list

git stash apply

git stash pop
